נספח 1 – סכמת בסיס הנתונים הגולמיים
CREATE TABLE "knesset_speeches" (
    "id" INTEGER NOT NULL CONSTRAINT "PK_knesset_speeches" PRIMARY KEY,
    "text" TEXT NULL,
    "uuid" TEXT NOT NULL,
    "knesset" INTEGER NOT NULL,
    "session_number" INTEGER NULL,
    "date" TEXT NOT NULL,
    "person_id" INTEGER NULL,
    "name_id" INTEGER NULL,
    "chair" INTEGER NOT NULL,
    "topic_id" INTEGER NULL,
    "topic_extra_id" INTEGER NULL,
    "qa" INTEGER NULL,
    "query" TEXT NULL,
    "only_read" INTEGER NULL,
    CONSTRAINT "FK_knesset_speeches_names_name_id" FOREIGN KEY ("name_id") REFERENCES "names" ("id"),
    CONSTRAINT "FK_knesset_speeches_topic_extras_topic_extra_id" FOREIGN KEY ("topic_extra_id") REFERENCES "topic_extras" ("id"),
    CONSTRAINT "FK_knesset_speeches_topics_topic_id" FOREIGN KEY ("topic_id") REFERENCES "topics" ("id")
)
CREATE TABLE "names" (
    "id" INTEGER NOT NULL CONSTRAINT "PK_names" PRIMARY KEY AUTOINCREMENT,
    "name" TEXT NULL
)
CREATE TABLE "people" (
    "id" INTEGER NOT NULL CONSTRAINT "PK_people" PRIMARY KEY,
    "person_id" INTEGER NULL,
    "start_date" TEXT NULL,
    "end_date" TEXT NULL,
    "knesset" INTEGER NULL,
    "first_name" TEXT NOT NULL,
    "surname" TEXT NOT NULL,
    "gender" INTEGER NOT NULL,
    "faction_id" INTEGER NULL,
    "faction" TEXT NULL,
    "party_name" TEXT NULL,
    "dob" TEXT NULL,
    "cob" TEXT NULL,
    "yod" INTEGER NULL,
    "yoi" INTEGER NULL,
    "city" TEXT NULL,
    "languages" TEXT NULL
)
CREATE TABLE sqlite_sequence(name,seq)
CREATE TABLE "topic_extras" (
    "id" INTEGER NOT NULL CONSTRAINT "PK_topic_extras" PRIMARY KEY AUTOINCREMENT,
    "topic_extra" TEXT NULL
)
CREATE TABLE "topics" (
    "id" INTEGER NOT NULL CONSTRAINT "PK_topics" PRIMARY KEY AUTOINCREMENT,
    "topic" TEXT NULL
)
CREATE VIEW knesset_speeches_view AS
SELECT
	k.id,
	n.name,
	k.text,
	k.knesset,
	k.session_number,
	k.date,
	t.topic,
	te.topic_extra,
	k.person_id,
	k.name_id,
	k.chair,
	k.topic_id,
	k.topic_extra_id,
	k.qa,
	k.query,
	k.only_read,
	k.uuid
FROM knesset_speeches AS k
LEFT JOIN topics AS t ON k.topic_id = t.id
LEFT JOIN topic_extras AS te ON k.topic_extra_id = te.id
LEFT JOIN names AS n ON k.name_id = n.id
